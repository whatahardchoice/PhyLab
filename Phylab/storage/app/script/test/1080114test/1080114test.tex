\subsubsection*{(2)数据处理}
用逐差法计算条纹间距${\Delta}x$:
$$\overline{{\Delta}x} = \displaystyle\frac{\sum\limits_{i=1}^{10}{\mid}x_{i+10}-x_i{\mid}}{10{\times}10} = 0.327mm$$

计算波长$\lambda$:
$$\bar{b} = \displaystyle\frac{b_\text{正}+b_\text{反}}{2} = \displaystyle\frac{ (4.638-4.034)+(4.636-3.991)}{2} = 0.6245mm$$

$$\bar{b'} = \displaystyle\frac{b'_\text{正}+b'_\text{反}}{2} = \displaystyle\frac{ (5.609-3.839)+(5.625-3.895)}{2} = 1.75mm$$

$$ S ={\mid} 115.34 - 85.94 {\mid}= 29.4cm$$

$$ S' ={\mid} 115.34 - 63.39 {\mid}= 51.95cm$$

$$ {\lambda} = \displaystyle\frac{{\Delta}x\sqrt{bb'}}{S+S'} = 420.2nm$$

\subsubsection*{(3)不确定度计算}
${\triangle}x$的不确定度：

$10{\triangle}x$的A类不确定度：
$$u_a(10{\triangle}x) = \sqrt{\displaystyle\frac{\sum\limits_{i=1}^{10} (10{{\triangle}x}_i-{10\overline{{\triangle}x}})^2}{10{\times}(10-1)}} = 0.02056mm$$
$10{\triangle}x$的B类不确定度：
$$u_b({10{\triangle}x})=\displaystyle\frac{\bigtriangleup\text{仪}}{\sqrt{3}}
= \frac{0.01}{2{\times}\sqrt{3}} = 0.00289mm$$
$10{\triangle}x$的不确定度：
$$u(10{\triangle}x)=\sqrt{{u_a(10{\triangle}x}^2)+{u_b(10{\triangle}x}^2)} = 0.02076mm$$
${\triangle}x$的不确定度：
$${\therefore}u({\triangle}x) = \displaystyle\frac{u(10{\triangle}x)}{10} = 0.002076mm$$

$$ \displaystyle\frac{{\Delta}b}{b} = \displaystyle\frac{{\Delta}b'}{b} = 0.025 $$

$b'$的不确定度：
$$ u(b') = \displaystyle\frac{ 0.6245{\times}0.025}{\sqrt{3}} =0.009014mm$$
b的不确定度：
$$ u(b) = \displaystyle\frac{ 1.75{\times}0.025}{\sqrt{3}} =0.02526mm$$
S的不确定度：
$${\Delta}S = {\Delta}S' = 0.5cm$$
$$ u(S+S') = \sqrt{2} {\times} 0.289 = 0.409cm$$

不确定度的合成：
$$ \ln{\lambda} = \ln{{\Delta}x}+ \displaystyle\frac{1}{2}(\ln{b}+\ln{b'}) - \ln(S+S')$$

$$\displaystyle\frac{\ln{\lambda}}{\lambda} = \displaystyle\frac{\ln{\Delta}x}{{\Delta}x}+\displaystyle\frac{1}{2}(\displaystyle\frac{\ln{b}}{b}+\displaystyle\frac{\ln{b'}}{b'}) - \displaystyle\frac{\ln{(S+S')}}{S+S'} $$

$$\displaystyle\frac{u({\lambda})}{\lambda} = \sqrt{ [{\frac{u({\Delta}x)}{{\Delta}x}}]^2+\frac{1}{4}[{\frac{u(b)}{b}}]^2 +\frac{1}{4}[{\frac{u(b')}{b'}}]^2 + [\frac{u(S+S')}{S+S'}]^2   } = 0.008093$$

$$ u({\lambda}) = 3.401 nm$$
最终结果为：
$$ {\lambda}{\pm}{u({\lambda})} = 420 {\pm} 4 nm $$
